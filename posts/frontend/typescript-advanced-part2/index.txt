```typescript
// 기본 엔티티
interface User {
  id: number;
  name: string;
  email: string;
  role: 'user' | 'admin' | 'moderator';
  createdAt: string;
  updatedAt: string;
  lastLoginAt: string | null;
}

// Create: id와 타임스탬프 제외
type CreateUserDTO = Omit<User, 'id' | 'createdAt' | 'updatedAt' | 'lastLoginAt'>;

// Update: 모든 필드 선택적 (id 제외)
type UpdateUserDTO = Partial<Omit<User, 'id' | 'createdAt'>>;

// List: 목록에 필요한 필드만
type UserListItem = Pick<User, 'id' | 'name' | 'email' | 'role'>;

// Detail: 전체 정보
type UserDetail = User;

// API 함수들
async function getUsers(
  page: number = 1,
  pageSize: number = 20
): Promise<PaginatedResponse<UserListItem>> {
  const response = await fetch(`/api/users?page=${page}&pageSize=${pageSize}`);
  return response.json();
}

async function getUser(id: number): Promise<UserDetail> {
  const response = await fetch(`/api/users/${id}`);
  const result: ApiResponse<UserDetail> = await response.json();

  if (!result.success || !result.data) {
    throw new Error(result.error?.message || 'Failed to fetch user');
  }

  return result.data;
}

async function createUser(data: CreateUserDTO): Promise<User> {
  const response = await fetch('/api/users', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });

  const result: ApiResponse<User> = await response.json();

  if (!result.success || !result.data) {
    throw new Error(result.error?.message || 'Failed to create user');
  }

  return result.data;
}

async function updateUser(id: number, data: UpdateUserDTO): Promise<User> {
  const response = await fetch(`/api/users/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data)
  });

  const result: ApiResponse<User> = await response.json();

  if (!result.success || !result.data) {
    throw new Error(result.error?.message || 'Failed to update user');
  }

  return result.data;
}

async function deleteUser(id: number): Promise<void> {
  const response = await fetch(`/api/users/${id}`, {
    method: 'DELETE'
  });

  const result: ApiResponse<void> = await response.json();

  if (!result.success) {
    throw new Error(result.error?.message || 'Failed to delete user');
  }
}
```

```typescript
// 에러 타입 정의
class ApiError extends Error {
  constructor(
    public statusCode: number,
    message: string,
    public code?: string,
    public details?: Record<string, unknown>
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

// 공통 fetch 
async function apiFetch<T>(
  url: string,
  options?: RequestInit
): Promise<T> {
  try {
    const response = await fetch(url, options);

    if (!response.ok) {
      const error: ApiResponse<never> = await response.json();
      throw new ApiError(
        response.status,
        error.error?.message || 'API request failed',
        error.error?.code,
        error.error?.details
      );
    }

    const result: ApiResponse<T> = await response.json();

    if (!result.success || !result.data) {
      throw new ApiError(
        500,
        result.error?.message || 'Invalid response',
        result.error?.code
      );
    }

    return result.data;
  } catch (error) {
    if (error instanceof ApiError) {
      throw error;
    }

    throw new ApiError(
      500,
      error instanceof Error ? error.message : 'Unknown error'
    );
  }
}

// 사용
try {
  const user = await apiFetch<User>('/api/users/1');
  console.log(user.name);
} catch (error) {
  if (error instanceof ApiError) {
    console.error(`API Error [${error.code}]:`, error.message);
    console.error('Status:', error.statusCode);
    console.error('Details:', error.details);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

```typescript
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string | number;
  emptyMessage?: string;
  loading?: boolean;
}

function List<T>({
  items,
  renderItem,
  keyExtractor,
  emptyMessage = 'No items',
  loading = false
}: ListProps<T>) {
  if (loading) {
    return <div>Loading...</div>;
  }

  if (items.length === 0) {
    return <div>{emptyMessage}</div>;
  }

  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item)}>
          {renderItem(item, index)}
        </li>
      ))}
    </ul>
  );
}

// 사용
interface User {
  id: number;
  name: string;
  email: string;
}

function UserList({ users }: { users: User[] }) {
  return (
    <List
      items={users}
      renderItem={(user) => (
        <div>
          <h3>{user.name}</h3>
          <p>{user.email}</p>
        </div>
      )}
      keyExtractor={(user) => user.id}
      emptyMessage="No users found"
    />
  );
}
```

```typescript
// 폼 값 정의
interface FormValues {
  name: string;
  email: string;
  age: number;
  terms: boolean;
}

// 에러 타입 (모든 필드 선택적)
type FormErrors = Partial<Record<keyof FormValues, string>>;

// 또는 더 명시적으로
interface FormErrors {
  name?: string;
  email?: string;
  age?: string;
  terms?: string;
}

// 폼 Props
interface FormProps {
  initialValues?: Partial<FormValues>;
  onSubmit: (values: FormValues) => void | Promise<void>;
  validate?: (values: FormValues) => FormErrors;
}

function UserForm({ initialValues, onSubmit, validate }: FormProps) {
  const [values, setValues] = React.useState<FormValues>({
    name: '',
    email: '',
    age: 0,
    terms: false,
    ...initialValues
  });

  const [errors, setErrors] = React.useState<FormErrors>({});
  const [isSubmitting, setIsSubmitting] = React.useState(false);

  const handleChange = <K extends keyof FormValues>(
    field: K,
    value: FormValues[K]
  ) => {
    setValues(prev => ({ ...prev, [field]: value }));
    // 에러 클리어
    setErrors(prev => ({ ...prev, [field]: undefined }));
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    // 검증
    if (validate) {
      const validationErrors = validate(values);
      if (Object.keys(validationErrors).length > 0) {
        setErrors(validationErrors);
        return;
      }
    }

    setIsSubmitting(true);
    try {
      await onSubmit(values);
    } catch (error) {
      console.error(error);
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <input
          type="text"
          value={values.name}
          onChange={(e) => handleChange('name', e.target.value)}
        />
        {errors.name && <span>{errors.name}</span>}
      </div>

      <div>
        <input
          type="email"
          value={values.email}
          onChange={(e) => handleChange('email', e.target.value)}
        />
        {errors.email && <span>{errors.email}</span>}
      </div>

      <div>
        <input
          type="number"
          value={values.age}
          onChange={(e) => handleChange('age', Number(e.target.value))}
        />
        {errors.age && <span>{errors.age}</span>}
      </div>

      <div>
        <input
          type="checkbox"
          checked={values.terms}
          onChange={(e) => handleChange('terms', e.target.checked)}
        />
        {errors.terms && <span>{errors.terms}</span>}
      </div>

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? 'Submitting...' : 'Submit'}
      </button>
    </form>
  );
}

// 사용
function App() {
  const handleSubmit = async (values: FormValues) => {
    console.log(values);
    await createUser(values);
  };

  const validate = (values: FormValues): FormErrors => {
    const errors: FormErrors = {};

    if (!values.name) {
      errors.name = 'Name is required';
    }

    if (!values.email) {
      errors.email = 'Email is required';
    } else if (!/\S+@\S+\.\S+/.test(values.email)) {
      errors.email = 'Email is invalid';
    }

    if (values.age < 18) {
      errors.age = 'Must be 18 or older';
    }

    if (!values.terms) {
      errors.terms = 'You must accept the terms';
    }

    return errors;
  };

  return <UserForm onSubmit={handleSubmit} validate={validate} />;
}
```

```typescript
// Context 타입 정의
interface AuthContextType {
  user: User | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
  updateProfile: (updates: Partial<User>) => Promise<void>;
}

// Context 생성 - undefined 허용
const AuthContext = React.createContext<AuthContextType | undefined>(
  undefined
);

// Provider 컴포넌트
interface AuthProviderProps {
  children: React.ReactNode;
}

function AuthProvider({ children }: AuthProviderProps) {
  const [user, setUser] = React.useState<User | null>(null);
  const [isLoading, setIsLoading] = React.useState(true);

  const login = async (email: string, password: string) => {
    setIsLoading(true);
    try {
      const user = await loginAPI(email, password);
      setUser(user);
    } finally {
      setIsLoading(false);
    }
  };

  const logout = () => {
    setUser(null);
  };

  const updateProfile = async (updates: Partial<User>) => {
    if (!user) throw new Error('No user logged in');
    const updated = await updateUserAPI(user.id, updates);
    setUser(updated);
  };

  const value: AuthContextType = {
    user,
    isAuthenticated: user !== null,
    isLoading,
    login,
    logout,
    updateProfile
  };

  return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>;
}

// Custom Hook
function useAuth(): AuthContextType {
  const context = React.useContext(AuthContext);

  if (context === undefined) {
    throw new Error('useAuth must be used within AuthProvider');
  }

  return context;
}
```

```typescript
import create from 'zustand';


interface StoreState {

  count: number;
  user: User | null;
  theme: 'light' | 'dark';

  // 액션
  increment: () => void;
  decrement: () => void;
  reset: () => void;
  setUser: (user: User | null) => void;
  toggleTheme: () => void;
}

// Store 생성
const useStore = create<StoreState>((set) => ({
  // 초기 상태
  count: 0,
  user: null,
  theme: 'light',

  // 액션
  increment: () => set((state) => ({ count: state.count + 1 })),
  decrement: () => set((state) => ({ count: state.count - 1 })),
  reset: () => set({ count: 0 }),
  setUser: (user) => set({ user }),
  toggleTheme: () =>
    set((state) => ({
      theme: state.theme === 'light' ? 'dark' : 'light'
    }))
}));

// 사용
function Counter() {
  const count = useStore((state) => state.count);
  const increment = useStore((state) => state.increment);
  const decrement = useStore((state) => state.decrement);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={increment}>+</button>
      <button onClick={decrement}>-</button>
    </div>
  );
}
```

### Redux Toolkit 패턴

```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit';

// 상태 타입
interface UserState {
  currentUser: User | null;
  users: User[];
  loading: boolean;
  error: string | null;
}

// 초기 상태
const initialState: UserState = {
  currentUser: null,
  users: [],
  loading: false,
  error: null
};

// Slice 생성
const userSlice = createSlice({
  name: 'user',
  initialState,
  reducers: {
    setLoading: (state, action: PayloadAction<boolean>) => {
      state.loading = action.payload;
    },

    setError: (state, action: PayloadAction<string | null>) => {
      state.error = action.payload;
    },

    setCurrentUser: (state, action: PayloadAction<User | null>) => {
      state.currentUser = action.payload;
    },

    setUsers: (state, action: PayloadAction<User[]>) => {
      state.users = action.payload;
    },

    addUser: (state, action: PayloadAction<User>) => {
      state.users.push(action.payload);
    },

    updateUser: (
      state,
      action: PayloadAction<{ id: number; updates: Partial<User> }>
    ) => {
      const { id, updates } = action.payload;
      const user = state.users.find((u) => u.id === id);
      if (user) {
        Object.assign(user, updates);
      }
    },

    deleteUser: (state, action: PayloadAction<number>) => {
      state.users = state.users.filter((u) => u.id !== action.payload);
    }
  }
});


export const {
  setLoading,
  setError,
  setCurrentUser,
  setUsers,
  addUser,
  updateUser,
  deleteUser
} = userSlice.actions;


export default userSlice.reducer;

// Thunk 예시
export const fetchUsers = () => async (dispatch: AppDispatch) => {
  dispatch(setLoading(true));
  try {
    const users = await getUsersAPI();
    dispatch(setUsers(users));
  } catch (error) {
    dispatch(setError(error instanceof Error ? error.message : 'Unknown error'));
  } finally {
    dispatch(setLoading(false));
  }
};
```

## 고급 타입 패턴

### Mapped Types

객체의 모든 키를 순회하며 새로운 타입을 만듭니다.

```typescript
// 모든 속성을 선택적으로
type Optional<T> = {
  [P in keyof T]?: T[P];
};

// 모든 속성을 nullable로
type Nullable<T> = {
  [P in keyof T]: T[P] | null;
};

// 특정 키만 선택적으로
type PartialBy<T, K extends keyof T> = Omit<T, K> & Partial<Pick<T, K>>;

interface User {
  id: number;
  name: string;
  email: string;
  age: number;
}

// name과 email만 선택적
type UserWithOptionalContact = PartialBy<User, 'name' | 'email'>;
// { id: number; age: number; name?: string; email?: string }

// 함수 타입을 Promise로 변환
type Promisify<T> = {
  [P in keyof T]: T[P] extends (...args: infer A) => infer R
    ? (...args: A) => Promise<R>
    : T[P];
};

interface SyncAPI {
  getUser: (id: number) => User;
  deleteUser: (id: number) => void;
  count: number;
}

type AsyncAPI = Promisify<SyncAPI>;
// {
//   getUser: (id: number) => Promise<User>;
//   deleteUser: (id: number) => Promise<void>;
//   count: number;
// }
```

### Conditional Types

조건에 따라 다른 타입을 반환합니다.

```typescript
// 기본 조건부 타입
type IsString<T> = T extends string ? true : false;

type A = IsString<string>; // true
type B = IsString<number>; // false

// 실용적 예시: NonNullable
type NonNullable<T> = T extends null | undefined ? never : T;

type T1 = NonNullable<string | null>; // string
type T2 = NonNullable<number | undefined>; // number

// 중첩 조건부 타입
type TypeName<T> =
  T extends string ? 'string' :
  T extends number ? 'number' :
  T extends boolean ? 'boolean' :
  T extends undefined ? 'undefined' :
  T extends Function ? 'function' :
  'object';

type T3 = TypeName<string>; // 'string'
type T4 = TypeName<() => void>; // 'function'
type T5 = TypeName<{}>; // 'object'

// Flatten 타입 (배열을 펼침)
type Flatten<T> = T extends Array<infer U> ? U : T;

type T6 = Flatten<string[]>; // string
type T7 = Flatten<number>; // number

// 깊은 readonly
type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? DeepReadonly<T[P]>
    : T[P];
};

interface User {
  id: number;
  profile: {
    name: string;
    settings: {
      theme: string;
    };
  };
}

type ReadonlyUser = DeepReadonly<User>;
// {
//   readonly id: number;
//   readonly profile: {
//     readonly name: string;
//     readonly settings: {
//       readonly theme: string;
//     };
//   };
```
