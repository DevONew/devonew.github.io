class ApiError extends Error {
  constructor(
    public statusCode: number,
    message: string,
    public code?: string,
    public details?: Record<string, unknown>
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

// ê³µí†µ fetch 
async function apiFetch<T>(
  url: string,
  options?: RequestInit
): Promise<T> {
  try {
    const response = await fetch(url, options);

    if (!response.ok) {
      const error: ApiResponse<never> = await response.json();
      throw new ApiError(
        response.status,
        error.error?.message || 'API request failed',
        error.error?.code,
        error.error?.details
      );
    }

    const result: ApiResponse<T> = await response.json();

    if (!result.success || !result.data) {
      throw new ApiError(
        500,
        result.error?.message || 'Invalid response',
        result.error?.code
      );
    }

    return result.data;
  } catch (error) {
    if (error instanceof ApiError) {
      throw error;
    }

    throw new ApiError(
      500,
      error instanceof Error ? error.message : 'Unknown error'
    );
  }
}

// ì‚¬ìš©
try {
  const user = await apiFetch<User>('/api/users/1');
  console.log(user.name);
} catch (error) {
  if (error instanceof ApiError) {
    console.error(`API Error [${error.code}]:`, error.message);
    console.error('Status:', error.statusCode);
    console.error('Details:', error.details);
  } else {
    console.error('Unexpected error:', error);
  }
}

interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string | number;
  emptyMessage?: string;
  loading?: boolean;
}

function List<T>({
  items,
  renderItem,
  keyExtractor,
  emptyMessage = 'No items',
  loading = false
}: ListProps<T>) {
  if (loading) {
    return <div>Loading...</div>;
  }

  if (items.length === 0) {
    return <div>{emptyMessage}</div>;
  }

  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item)}>
          {renderItem(item, index)}
        </li>
      ))}
    </ul>
  );
}

  const [isLoading, setIsLoading] = React.useState(true);

  const login = async (email: string, password: string) => {
    setIsLoading(true);
    try {
      const user = await loginAPI(email, password);
      setUser(user);
    } finally {
      setIsLoading(false);
    }
  };

  const logout = () => {
    setUser(null);
  };

  const updateProfile = async (updates: Partial<User>) => {
    if (!user) throw new Error('No user logged in');
    const updated = await updateUserAPI(user.id, updates);
    setUser(updated);
  };

  const value: AuthContextType = {
    user,
    isAuthenticated: user !== null,
    isLoading,
    login,
    logout,
    updateProfile
  };

  return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>;
}

// Custom Hook
function useAuth(): AuthContextType {
  const context = React.useContext(AuthContext);

  if (context === undefined) {
    throw new Error('useAuth must be used within AuthProvider');
  }

  return context;
}

// ì‚¬ìš©
function ProfilePage() {
  const { user, updateProfile, logout } = useAuth();

  if (!user) {
    return <div>Please login</div>;
  }

  return (
    <div>
      <h1>{user.name}</h1>
      <button onClick={() => updateProfile({ name: 'New Name' })}>
        Update Name
      </button>
      <button onClick={logout}>Logout</button>
    </div>
  );
}

    },

    },

      state,
      action: PayloadAction<{ id: number; updates: Partial<User> }>
    ) => {

      const user = state.users.find((u) => u.id === id);
      if (user) {
        Object.assign(user, updates);
      }
    },

      state.users = state.users.filter((u) => u.id !== action.payload);
    }
  }

  setError,
  setCurrentUser,
  setUsers,
