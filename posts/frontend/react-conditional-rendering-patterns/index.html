type ConditionalComponent = {
  <T>(props: { children: React.ReactNode }): JSX.Element;
  When: <T>(props: {
    condition: boolean;
    children: React.ReactNode
  }) => JSX.Element | null;
  Else: (props: { children: React.ReactNode }) => JSX.Element;
};

const Conditional: ConditionalComponent = ({ children }) => {
  const childrenArray = React.Children.toArray(children);

  for (const child of childrenArray) {
    if (React.isValidElement(child) && child.type === Conditional.When) {
      if (child.props.condition) {
        return child;
      }
    }
  }

  const elseChild = childrenArray.find(
    child => React.isValidElement(child) && child.type === Conditional.Else
  );

  return elseChild ? (elseChild as React.ReactElement) : null;
};

Conditional.When = ({ condition, children }) => {
  return condition ? <>{children}</> : null;
};

Conditional.Else = ({ children }) => <>{children}</>;
