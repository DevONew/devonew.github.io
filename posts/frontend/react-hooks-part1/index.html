function Counter() {
  const [count, setCount] = useState(0);

  // лӮҳмҒң мҳҲ: мқҙм „ мғҒнғңм—җ мқҳмЎҙн•ҳлҠ” кІҪмҡ°
  const handleIncrement = () => {
    setCount(count + 1);
    setCount(count + 1); // л‘җ лІҲ мҰқк°Җ? м•„лӢҲмҡ”, н•ң лІҲл§Ң мҰқк°Җ!
  };

  // мўӢмқҖ мҳҲ: н•ЁмҲҳнҳ• м—…лҚ°мқҙнҠё
  const handleIncrementTwice = () => {
    setCount(prev => prev + 1);
    setCount(prev => prev + 1); // м ңлҢҖлЎң л‘җ лІҲ мҰқк°Җ
  };

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={handleIncrement}>+1 (мһҳлӘ»лҗң л°©лІ•)</button>
      <button onClick={handleIncrementTwice}>+2 (мҳ¬л°”лҘё л°©лІ•)</button>
    </div>
  );
}

  • мқҙм „ мғҒнғңлҘј кё°л°ҳмңјлЎң м—…лҚ°мқҙнҠён•  л•Ң м•Ҳм „

к°қмІҙ мғҒнғң кҙҖлҰ¬

к°қмІҙ нғҖмһ…мқҳ мғҒнғңлҘј м—…лҚ°мқҙнҠён•  л•ҢлҠ” мЈјмқҳк°Җ н•„мҡ”н•©лӢҲлӢӨ. ReactлҠ” м–•мқҖ л№„көҗ(shallow comparison)лҘј н•ҳкё° л•Ңл¬ём—җ к°қмІҙмқҳ мқјл¶Җл§Ң ліҖкІҪн•ҳл©ҙ лӮҳлЁём§Җ н•„л“ңк°Җ мӮ¬лқјм§‘лӢҲлӢӨ.

function UserForm() {
  const [user, setUser] = useState({
    name: '',
    email: '',
    age: 0
  });

  // лӮҳмҒң мҳҲ: мқјл¶Җл§Ң м—…лҚ°мқҙнҠён•ҳл©ҙ лӮҳлЁём§Җ мӮ¬лқјм§җ
  const handleNameChange = (e) => {
    setUser({ name: e.target.value }); // emailкіј ageк°Җ мӮ¬лқјм§җ!
  };

  // мўӢмқҖ мҳҲ: spread м—°мӮ°мһҗлЎң лі‘н•©
  const handleChange = (field) => (e) => {
    setUser(prev => ({
      ...prev,
      [field]: e.target.value
    }));
  };

  return (
    <form>
      <input
        value={user.name}
        onChange={handleChange('name')}
        placeholder="Name"
      />
      <input
        value={user.email}
        onChange={handleChange('email')}
        placeholder="Email"
      />
      <input
        type="number"
        value={user.age}
        onChange={handleChange('age')}
        placeholder="Age"
      />
      <p>{JSON.stringify(user, null, 2)}</p>
    </form>
  );
}

л°°м—ҙ мғҒнғң кҙҖлҰ¬

л°°м—ҙ мғҒнғңлҘј лӢӨлЈ° л•ҢлҠ” мӣҗліёмқ„ м§Ғм ‘ мҲҳм •н•ҳм§Җ л§җкі , мғҲлЎңмҡҙ л°°м—ҙмқ„ л§Ңл“Өм–ҙ л°ҳнҷҳн•ҙм•ј н•©лӢҲлӢӨ. мӢӨл¬ҙм—җм„ң к°ҖмһҘ л§Һмқҙ мӮ¬мҡ©н•ҳлҠ” CRUD нҢЁн„ҙмқ„ мӮҙнҺҙліҙкІ мҠөлӢҲлӢӨ.

function TodoList() {
  const [todos, setTodos] = useState([]);
  const [input, setInput] = useState('');

  // м¶”к°Җ
  const addTodo = () => {
    if (!input.trim()) return;
    setTodos(prev => [...prev, {
      id: Date.now(),
      text: input,
      completed: false
    }]);
    setInput('');
  };

  // нҶ кёҖ
  const toggleTodo = (id) => {
    setTodos(prev => prev.map(todo =>
      todo.id === id
        ? { ...todo, completed: !todo.completed }
        : todo
    ));
  };

  // мӮӯм ң
  const deleteTodo = (id) => {
    setTodos(prev => prev.filter(todo => todo.id !== id));
  };

  // мҲҳм •
  const updateTodo = (id, newText) => {
    setTodos(prev => prev.map(todo =>
      todo.id === id
        ? { ...todo, text: newText }
        : todo
    ));
  };

  return (
    <div>
      <input
        value={input}
        onChange={(e) => setInput(e.target.value)}
        onKeyPress={(e) => e.key === 'Enter' && addTodo()}
      />
      <button onClick={addTodo}>Add</button>
      <ul>
        {todos.map(todo => (
          <li key={todo.id}>
            <input
              type="checkbox"
              checked={todo.completed}
              onChange={() => toggleTodo(todo.id)}
            />
            <span
              style={{
                textDecoration: todo.completed ? 'line-through' : 'none'
              }}
            >
              {todo.text}
            </span>
            <button onClick={() => deleteTodo(todo.id)}>Delete</button>
          </li>
        ))}
      </ul>
    </div>
  );
}

function Timer() {
  const [seconds, setSeconds] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setSeconds(s => s + 1);
    }, 1000);

    // нҒҙлҰ°м—… н•ЁмҲҳ: м–ёл§ҲмҡҙнҠё мӢң лҳҗлҠ” лӢӨмқҢ effect мӢӨн–ү м „м—җ нҳём¶ң
    return () => {
      clearInterval(interval);
    };
  }, []); // л№Ҳ л°°м—ҙмқҙлҜҖлЎң л§ҲмҡҙнҠё/м–ёл§ҲмҡҙнҠё мӢңм—җл§Ң

  return <div>Seconds: {seconds}</div>;
}

API нҳём¶ңмқҖ useEffectм—җм„ң мһҗмЈј мӮ¬мҡ©лҗҳлҠ” нҢЁн„ҙмһ…лӢҲлӢӨ. н•ҳм§Җл§Ң м»ҙнҸ¬л„ҢнҠёк°Җ м–ёл§ҲмҡҙнҠёлҗҳкұ°лӮҳ userIdк°Җ ліҖкІҪлҗҳлҠ” мғҒнҷ©мқ„ мІҳлҰ¬н•ҙм•ј н•©лӢҲлӢӨ.

function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    // cleanupмқ„ мң„н•ң flag
    let cancelled = false;

    async function fetchUser() {
      try {
        setLoading(true);
        setError(null);
        const response = await fetch(`/api/users/${userId}`);
        const data = await response.json();

        if (!cancelled) {
          setUser(data);
        }
      } catch (err) {
        if (!cancelled) {
          setError(err.message);
        }
      } finally {
        if (!cancelled) {
          setLoading(false);
        }
      }
    }

    fetchUser();

    // нҒҙлҰ°м—…: м»ҙнҸ¬л„ҢнҠё м–ёл§ҲмҡҙнҠёлӮҳ userId ліҖкІҪ мӢң
    return () => {
      cancelled = true;
    };
  }, [userId]);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!user) return <div>No user found</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}

function WindowSize() {
  const [size, setSize] = useState({
    width: window.innerWidth,
    height: window.innerHeight
  });

  useEffect(() => {
    const handleResize = () => {
      setSize({
        width: window.innerWidth,
        height: window.innerHeight
      });
    };

    window.addEventListener('resize', handleResize);

    // нҒҙлҰ°м—…
    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, []);

  return (
    <div>
      Width: {size.width}px, Height: {size.height}px
    </div>
  );
}

function useLocalStorage(key, initialValue) {
  // lazy initializationмңјлЎң localStorage мқҪкё°
  const [value, setValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(error);
      return initialValue;
    }
  });

  // valueк°Җ ліҖкІҪлҗ  л•Ң localStorageм—җ м ҖмһҘ
  useEffect(() => {
    try {
      window.localStorage.setItem(key, JSON.stringify(value));
    } catch (error) {
      console.error(error);
    }
  }, [key, value]);

  return [value, setValue];
}

      />
      <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>

}

import { useRef, useEffect } from 'react';

function AutoFocusInput() {

useRefмқҳ к°•л Ҙн•ң нҷңмҡ©лІ• мӨ‘ н•ҳлӮҳлҠ” 'мқҙм „ к°’'мқ„ кё°м–өн•ҳлҠ” кІғмһ…лӢҲлӢӨ. refлҠ” лҰ¬л ҢлҚ”л§Ғлҗҳм–ҙлҸ„ к°’мқҙ мң м§Җлҗҳкё° л•Ңл¬ём—җ к°ҖлҠҘн•©лӢҲлӢӨ.

function usePrevious(value) {
  const ref = useRef();

  useEffect(() => {
    ref.current = value;
  }, [value]);

  return ref.current;
}

// мӮ¬мҡ©
function Counter() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevious(count);

  return (

      <p>Previous: {prevCount}</p>
      <button onClick={() => setCount(c => c + 1)}>+</button>

  );
}

function Timer() {

  const countRef = useRef(count);

  // ref м—…лҚ°мқҙнҠёлҠ” лҰ¬л ҢлҚ”л§Ғмқ„ л°ңмғқмӢңнӮӨм§Җ м•ҠмқҢ

  return (

      <p>Count: {count}</p>

import { createContext, useContext, useState } from 'react';

// Context мғқм„ұ
const ThemeContext = createContext();

// Provider м»ҙнҸ¬л„ҢнҠё
function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');

  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

// HookмңјлЎң Context мӮ¬мҡ©
function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
}

// м»ҙнҸ¬л„ҢнҠём—җм„ң мӮ¬мҡ©
function ThemedButton() {
  const { theme, toggleTheme } = useTheme();

  return (
    <button
      onClick={toggleTheme}
      style={{
        background: theme === 'light' ? '#fff' : '#333',
        color: theme === 'light' ? '#000' : '#fff'
      }}
    >
      Toggle Theme (Current: {theme})
    </button>
  );
}

// App
function App() {
  return (
    <ThemeProvider>
      <ThemedButton />
    </ThemeProvider>
  );
}

мқёмҰқ Context

мӢӨл¬ҙм—җм„ң к°ҖмһҘ л§Һмқҙ мӮ¬мҡ©лҗҳлҠ” нҢЁн„ҙ мӨ‘ н•ҳлӮҳлҠ” мқёмҰқ(Authentication) Contextмһ…лӢҲлӢӨ. лЎңк·ёмқё мғҒнғңлҘј м „м—ӯм ҒмңјлЎң кҙҖлҰ¬н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

const AuthContext = createContext();

function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // мҙҲкё° мқёмҰқ мғҒнғң нҷ•мқё
    checkAuth();
  }, []);

  async function checkAuth() {
    try {
      const response = await fetch('/api/auth/me');
      if (response.ok) {
        const user = await response.json();
        setUser(user);
      }
    } catch (error) {
      console.error('Auth check failed:', error);
    } finally {
      setLoading(false);
    }
  }

  async function login(email, password) {
    const response = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, password })
    });

    if (response.ok) {
      const user = await response.json();
      setUser(user);
      return user;
    }

    throw new Error('Login failed');
  }

  async function logout() {
    await fetch('/api/auth/logout', { method: 'POST' });
    setUser(null);
  }

  const value = {
    user,
    loading,
    login,
    logout,
    isAuthenticated: !!user
  };

  return (
    <AuthContext.Provider value={value}>
      {children}
    </AuthContext.Provider>
  );
}

function useAuth() {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within AuthProvider');
  }
  return context;
}

// ліҙнҳёлҗң лқјмҡ°нҠё
function ProtectedRoute({ children }) {
  const { isAuthenticated, loading } = useAuth();

  if (loading) return <div>Loading...</div>;
  if (!isAuthenticated) return <Navigate to="/login" />;

  return children;
}
