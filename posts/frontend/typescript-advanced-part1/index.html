유틸리티 타입

TypeScript가 제공하는 내장 유틸리티 타입들은 기존 타입을 변환하여 새로운 타입을 만듭니다.

Partial<T>

Required<T>

반대도 있어야겠죠. 모든 속성을 필수로 만듭니다.

interface Config {
  host?: string;
  port?: number;
  timeout?: number;
  debug?: boolean;
}

// 모든 필드가 필수가 됨
type RequiredConfig = Required<Config>;
// {
//   host: string;
//   port: number;
//   timeout: number;
//   debug: boolean;
// }

// 기본값이 있어야함
const defaultConfig: Required<Config> = {
  host: 'localhost',
  port: 3000,
  timeout: 5000,
  debug: false
};

function createConfig(config: Config = {}): Required<Config> {
  return {
    host: config.host ?? defaultConfig.host,
    port: config.port ?? defaultConfig.port,
    timeout: config.timeout ?? defaultConfig.timeout,
    debug: config.debug ?? defaultConfig.debug
  };
}

// 사용
const config1 = createConfig(); // 모든 필드가 기본값
const config2 = createConfig({ port: 8080 }); // port만 변경, 나머지는 기본값

Extract<T, U>와 Exclude<T, U>

Union 타입에서 특정 타입을 추출하거나 제외합니다.

type T1 = 'a' | 'b' | 'c' | 'd';
type T2 = 'a' | 'c' | 'e';

// Extract: 교집합 
type Common = Extract<T1, T2>; // 'a' | 'c'

// Exclude: 차집합 
type Unique = Exclude<T1, T2>; // 'b' | 'd'

//  null/undefined 제거
type Value = string | number | null | undefined;
type NonNullableValue = Exclude<Value, null | undefined>; // string | number

// TypeScript 내장 NonNullable과 같음
type NonNullable<T> = Exclude<T, null | undefined>;

이벤트 타입 필터링

// 모든 이벤트 타입
type AllEvents = 'click' | 'scroll' | 'resize' | 'keydown' | 'keyup' | 'mousemove';

// 마우스 이벤트만 추출
type MouseEvent = Extract<AllEvents, 'click' | 'mousemove'>; // 'click' | 'mousemove'

// 마우스 이벤트 제외
type NonMouseEvent = Exclude<AllEvents, MouseEvent>;
// 'scroll' | 'resize' | 'keydown' | 'keyup'

// 함수 시그니처에서 특정 타입만 추출
type Fn1 = () => void;
type Fn2 = (x: number) => void;
type Fn3 = (x: string, y: number) => void;

type AllFn = Fn1 | Fn2 | Fn3;

// void를 반환하는 함수만 추출
type VoidFn = Extract<AllFn, (...args: any[]) => void>;
// Fn1 | Fn2 | Fn3 (모두 void 반환)

// 매개변수가 없는 함수만 추출
type NoArgsFn = Extract<AllFn, () => any>;
// Fn1

ReturnType<T>와 Parameters<T>

함수 타입에서 반환 타입과 매개변수 타입을 추출합니다.

함수 래퍼 (Wrapper)

// 로깅 추가
function withLogging<T extends (...args: any[]) => any>(fn: T) {
  return (...args: Parameters<T>): ReturnType<T> => {
    console.log(`[${fn.name}] Called with:`, args);
    const result = fn(...args);
    console.log(`[${fn.name}] Returned:`, result);
    return result;
  };
}

// 사용
const add = (a: number, b: number) => a + b;
const loggedAdd = withLogging(add);

loggedAdd(2, 3);
// [add] Called with: [2, 3]
// [add] Returned: 5

function memoize<T extends (...args: any[]) => any>(fn: T) {
  const cache = new Map<string, ReturnType<T>>();

  return (...args: Parameters<T>): ReturnType<T> => {
    const key = JSON.stringify(args);

    if (cache.has(key)) {
      console.log('Cache hit!');
      return cache.get(key)!;
    }

    const result = fn(...args);
    cache.set(key, result);
    return result;
  };
}

// 사용
const expensiveCalculation = (n: number) => {
  console.log('Computing...');
  return n * n;
};

const memoized = memoize(expensiveCalculation);

memoized(5); // "Computing..." >25
memoized(5); // "Cache hit!" >25

any

any는 TypeScript의 타입 시스템을 완전히 우회하는 역할을 합니다. 하지만 애니를 남발할 경우 TS를 사용하는 의미가 없다고 생각합니다.

any가 위험한 이유

let data: any = 'hello';

data.toUpperCase(); 
data.toFixed(); // 런타임 에러
data.nonExistentMethod(); // 런타임 에러
data.property.nested.deep.value; // 런타임 에러

// 컴파일에 잡히지 않는 버그
function processUser(user: any) {
  console.log(user.name.toUpperCase()); // user.name이 없으면
  console.log(user.age + 1); // age가 문자열이면

// 타입 추론 망가뜨림

unknown: 안전한 any

unknown은 any와 비슷하지만 타입 체크를 강제합니다.


let value: unknown = 'hello';

// 타입 체크 없이는 사용 불가
value.toUpperCase(); // 에러
value.length; // 에러
value(); // 에러

// 타입 체크 후 사용 가능
if (typeof value === 'string') {
  console.log(value.toUpperCase()); // 스트링으로 유추
}

// 여러 타입 체크
function processUnknown(value: unknown) {
  if (typeof value === 'string') {
    return value.toUpperCase();
  }

  if (typeof value === 'number') {
    return value.toFixed(2);
  }

  if (Array.isArray(value)) {
    return value.length;
  }

  return 'Unknown type';
}

any를 unknown으로 대체하기

// any 사용
function processDataBad(data: any) {
  return data.value * 2;
}

// unknown + 타입 가드
function processData(data: unknown): number {
  if (
    typeof data === 'object' &&
    data !== null &&
    'value' in data &&
    typeof (data as { value: unknown }).value === 'number'
  ) {
    return (data as { value: number }).value * 2;
  }

  throw new Error('Invalid data format');
}

// 또는 타입 가드 함수로 분리
interface DataWithValue {
  value: number;
}

function isDataWithValue(data: unknown): data is DataWithValue {
  return (
    typeof data === 'object' &&
    data !== null &&
    'value' in data &&
    typeof (data as DataWithValue).value === 'number'
  );
}

function processDataBetter(data: unknown): number {
  if (isDataWithValue(data)) {
    return data.value * 2; 
  }

  throw new Error('Invalid data format');
}

function mapObject<T extends Record<string, any>, U>(
  obj: T,
  fn: (value: T[keyof T]) => U
): Record<keyof T, U> {
  const result = {} as Record<keyof T, U>;
  for (const key in obj) {
    result[key] = fn(obj[key]);
  }
  return result;
}

const nums = { a: 1, b: 2, c: 3 };
const doubled = mapObject(nums, x => x * 2);
// Record<"a" | "b" | "c", number>
