class LinkedList {
  constructor() {
    this.head = null;
    this.size = 0;
  }

  // O(1) - ë§¨ ì•ì— ì‚½ì…
  prepend(data) {
    const newNode = new Node(data);
    newNode.next = this.head;
    this.head = newNode;
    this.size++;
  }

  // O(n) - ë§¨ ëì— ì‚½ì… (tail í¬ì¸í„° ì—†ì´)
  append(data) {
    const newNode = new Node(data);
    if (!this.head) {
      this.head = newNode;
      return;
    }
    let current = this.head;
    while (current.next) {
      current = current.next;
    }
    current.next = newNode;
    this.size++;
  }

  // O(1) - íŠ¹ì • ë…¸ë“œ ë‹¤ìŒì— ì‚½ì… (ë…¸ë“œë¥¼ ì•Œ ë•Œ)
  insertAfter(node, data) {
    const newNode = new Node(data);
    newNode.next = node.next;
    node.next = newNode;
    this.size++;
  }

  // O(1) - ë§¨ ì• ì‚­ì œ
  removeFirst() {
    if (!this.head) return null;
    const removed = this.head;
    this.head = this.head.next;
    this.size--;
    return removed.data;
  }
}

class BrowserHistory {
  constructor() {
    this.current = null;
  }

  visit(url) {
    const page = new DoublyNode(url);
    if (this.current) {
      this.current.next = page;
      page.prev = this.current;
    }
    this.current = page;
  }

  back() {
    if (this.current?.prev) {
      this.current = this.current.prev;
      return this.current.data;
    }
  }

  forward() {
    if (this.current?.next) {
      this.current = this.current.next;
      return this.current.data;
    }
  }
}

  const stack = [];
  const pairs = { '(': ')', '{': '}', '[': ']' };

  for (const char of str) {

  constructor() {
    this.front = null;
    this.rear = null;
    this.size = 0;

  // O(1) - ë’¤ì— ì¶”ê°€
  enqueue(data) {

    if (this.isEmpty()) {
      this.front = this.rear = newNode;

      this.rear = newNode;
    }
    this.size++;
  }

  // O(1) - ì•ì—ì„œ ì œê±°
  dequeue() {
    if (this.isEmpty()) return null;

    this.front = this.front.next;
    if (!this.front) this.rear = null;
    this.size--;
    return removed;
  }

  peek() {
    return this.front?.data;
  }

  isEmpty() {
    return this.size === 0;
  }
}
